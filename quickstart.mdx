# Quick Start Guide

Learn how to capture photos, record videos, and pick media from the gallery with the Camera Plugin.

## Basic Photo Capture

The simplest way to capture a photo is with a single method call:

<Tabs>
  <Tab title="PHP (Livewire/Blade)">
    ```php
    use Native\Mobile\Facades\Camera;
    use Native\Mobile\Attributes\OnNative;
    use Native\Mobile\Events\Camera\PhotoTaken;
    
    class PhotoComponent extends Component
    {
        public string $photoPath = '';
        
        public function capturePhoto()
        {
            // Launch camera to take a photo
            Camera::getPhoto();
        }
        
        #[OnNative(PhotoTaken::class)]
        public function handlePhotoTaken(string $path)
        {
            // Photo captured successfully
            $this->photoPath = $path;
            $this->processPhoto($path);
        }
        
        private function processPhoto(string $path)
        {
            // Upload to storage, process, etc.
        }
    }
    ```
  </Tab>
  
  <Tab title="JavaScript (Vue/React)">
    ```javascript
    import { Camera, On, Off, Events } from '#nativephp';
    import { ref, onMounted, onUnmounted } from 'vue';
    
    const photoPath = ref('');
    
    const capturePhoto = async () => {
        // Launch camera to take a photo
        await Camera.getPhoto();
    };
    
    const handlePhotoTaken = (payload) => {
        // Photo captured successfully
        photoPath.value = payload.path;
        processPhoto(payload.path);
    };
    
    const processPhoto = (path) => {
        // Upload to storage, process, etc.
        console.log('Photo saved at:', path);
    };
    
    onMounted(() => {
        On(Events.Camera.PhotoTaken, handlePhotoTaken);
    });
    
    onUnmounted(() => {
        Off(Events.Camera.PhotoTaken, handlePhotoTaken);
    });
    ```
  </Tab>
</Tabs>

## Recording Videos

Record videos with optional duration limits:

<Steps>
  <Step title="Basic Video Recording">
    <Tabs>
      <Tab title="PHP">
        ```php
        use Native\Mobile\Facades\Camera;
        
        // Simple video recording
        Camera::recordVideo();
        
        // With maximum duration (30 seconds)
        Camera::recordVideo(['maxDuration' => 30]);
        ```
      </Tab>
      
      <Tab title="JavaScript">
        ```javascript
        import { Camera } from '#nativephp';
        
        // Simple video recording
        await Camera.recordVideo();
        
        // With maximum duration (60 seconds)
        await Camera.recordVideo()
            .maxDuration(60);
        ```
      </Tab>
    </Tabs>
  </Step>
  
  <Step title="Using Fluent API">
    The fluent API allows method chaining for cleaner code:
    
    <Tabs>
      <Tab title="PHP">
        ```php
        use Native\Mobile\Facades\Camera;
        
        Camera::recordVideo()
            ->maxDuration(60)
            ->id('my-video-123')
            ->start();
        ```
      </Tab>
      
      <Tab title="JavaScript">
        ```javascript
        import { Camera } from '#nativephp';
        
        await Camera.recordVideo()
            .maxDuration(60)
            .id('my-video-123');
        ```
      </Tab>
    </Tabs>
  </Step>
  
  <Step title="Handle Video Events">
    <Tabs>
      <Tab title="PHP">
        ```php
        use Native\Mobile\Attributes\OnNative;
        use Native\Mobile\Events\Camera\VideoRecorded;
        use Native\Mobile\Events\Camera\VideoCancelled;
        
        #[OnNative(VideoRecorded::class)]
        public function handleVideoRecorded(string $path, string $mimeType, ?string $id)
        {
            // Video recorded successfully
            $this->videoPath = $path;
            $this->uploadVideo($path);
        }
        
        #[OnNative(VideoCancelled::class)]
        public function handleVideoCancelled(?string $id)
        {
            // User cancelled recording
            $this->notify('Video recording cancelled');
        }
        ```
      </Tab>
      
      <Tab title="JavaScript">
        ```javascript
        import { On, Off, Events } from '#nativephp';
        
        const handleVideoRecorded = (payload) => {
            console.log('Video path:', payload.path);
            console.log('MIME type:', payload.mimeType);
            console.log('ID:', payload.id);
            uploadVideo(payload.path);
        };
        
        const handleVideoCancelled = (payload) => {
            console.log('Recording cancelled, ID:', payload.id);
        };
        
        onMounted(() => {
            On(Events.Camera.VideoRecorded, handleVideoRecorded);
            On(Events.Camera.VideoCancelled, handleVideoCancelled);
        });
        
        onUnmounted(() => {
            Off(Events.Camera.VideoRecorded, handleVideoRecorded);
            Off(Events.Camera.VideoCancelled, handleVideoCancelled);
        });
        ```
      </Tab>
    </Tabs>
  </Step>
</Steps>

## Gallery Picker

Let users select existing photos or videos from their device:

<Tabs>
  <Tab title="PHP">
    ```php
    use Native\Mobile\Facades\Camera;
    use Native\Mobile\Attributes\OnNative;
    use Native\Mobile\Events\Gallery\MediaSelected;
    
    // Pick a single image
    Camera::pickImages('images', false);
    
    // Pick multiple images
    Camera::pickImages('images', true);
    
    // Pick any media type (images or videos)
    Camera::pickImages('all', true);
    
    #[OnNative(MediaSelected::class)]
    public function handleMediaSelected($success, $files, $count)
    {
        if ($success) {
            foreach ($files as $file) {
                $this->processMedia($file);
            }
        }
    }
    ```
  </Tab>
  
  <Tab title="JavaScript">
    ```javascript
    import { Camera, On, Events } from '#nativephp';
    
    // Pick a single image
    await Camera.pickImages()
        .images();
    
    // Pick multiple images (max 5)
    await Camera.pickImages()
        .images()
        .multiple()
        .maxItems(5);
    
    // Pick any media type
    await Camera.pickImages()
        .all()
        .multiple();
    
    const handleMediaSelected = (payload) => {
        if (payload.success) {
            console.log(`Selected ${payload.count} files`);
            payload.files.forEach(file => {
                console.log('Path:', file.path);
                console.log('Type:', file.mimeType);
                processMedia(file);
            });
        }
    };
    
    onMounted(() => {
        On(Events.Gallery.MediaSelected, handleMediaSelected);
    });
    ```
  </Tab>
</Tabs>

## Tracking Operations with IDs

Use optional IDs to track which operation triggered an event:

<Tabs>
  <Tab title="PHP">
    ```php
    use Native\Mobile\Facades\Camera;
    
    // Capture profile picture
    Camera::getPhoto(['id' => 'profile-pic']);
    
    // Capture cover photo
    Camera::getPhoto(['id' => 'cover-photo']);
    
    #[OnNative(PhotoTaken::class)]
    public function handlePhotoTaken(string $path, ?string $id)
    {
        if ($id === 'profile-pic') {
            $this->updateProfilePicture($path);
        } elseif ($id === 'cover-photo') {
            $this->updateCoverPhoto($path);
        }
    }
    ```
  </Tab>
  
  <Tab title="JavaScript">
    ```javascript
    import { Camera } from '#nativephp';
    
    // Capture profile picture
    await Camera.getPhoto()
        .id('profile-pic');
    
    // Capture cover photo
    await Camera.getPhoto()
        .id('cover-photo');
    
    const handlePhotoTaken = (payload) => {
        if (payload.id === 'profile-pic') {
            updateProfilePicture(payload.path);
        } else if (payload.id === 'cover-photo') {
            updateCoverPhoto(payload.path);
        }
    };
    ```
  </Tab>
</Tabs>

## Handling Permission Denials

Gracefully handle cases where the user denies camera permission:

<Tabs>
  <Tab title="PHP">
    ```php
    use Native\Mobile\Attributes\OnNative;
    use Native\Mobile\Events\Camera\PermissionDenied;
    
    #[OnNative(PermissionDenied::class)]
    public function handlePermissionDenied(string $action, ?string $id)
    {
        // Show user-friendly message
        $this->notify(
            'Camera permission is required to take photos. ' .
            'Please enable it in your device settings.'
        );
    }
    ```
  </Tab>
  
  <Tab title="JavaScript">
    ```javascript
    import { On, Events } from '#nativephp';
    
    const handlePermissionDenied = (payload) => {
        alert(
            'Camera permission is required. ' +
            'Please enable it in your device settings.'
        );
        console.log('Action:', payload.action);
        console.log('ID:', payload.id);
    };
    
    onMounted(() => {
        On(Events.Camera.PermissionDenied, handlePermissionDenied);
    });
    ```
  </Tab>
</Tabs>

<Note>
**Permissions are required!** Make sure you've enabled the `camera` permission in `config/nativephp.php` as described in the [Installation](/installation) guide.
</Note>

## Complete Example

Here's a complete example combining photos, videos, and gallery picker:

<Tabs>
  <Tab title="PHP">
    ```php
    <?php
    
    namespace App\Livewire;
    
    use Livewire\Component;
    use Native\Mobile\Facades\Camera;
    use Native\Mobile\Attributes\OnNative;
    use Native\Mobile\Events\Camera\PhotoTaken;
    use Native\Mobile\Events\Camera\VideoRecorded;
    use Native\Mobile\Events\Gallery\MediaSelected;
    
    class MediaCapture extends Component
    {
        public array $photos = [];
        public array $videos = [];
        
        public function takePhoto()
        {
            Camera::getPhoto();
        }
        
        public function recordVideo()
        {
            Camera::recordVideo()
                ->maxDuration(60)
                ->start();
        }
        
        public function pickFromGallery()
        {
            Camera::pickImages('all', true);
        }
        
        #[OnNative(PhotoTaken::class)]
        public function handlePhotoTaken(string $path)
        {
            $this->photos[] = $path;
        }
        
        #[OnNative(VideoRecorded::class)]
        public function handleVideoRecorded(string $path)
        {
            $this->videos[] = $path;
        }
        
        #[OnNative(MediaSelected::class)]
        public function handleMediaSelected($success, $files, $count)
        {
            if ($success) {
                foreach ($files as $file) {
                    if ($file['type'] === 'image') {
                        $this->photos[] = $file['path'];
                    } else {
                        $this->videos[] = $file['path'];
                    }
                }
            }
        }
        
        public function render()
        {
            return view('livewire.media-capture');
        }
    }
    ```
  </Tab>
  
  <Tab title="JavaScript (Vue)">
    ```vue
    <script setup>
    import { ref, onMounted, onUnmounted } from 'vue';
    import { Camera, On, Off, Events } from '#nativephp';
    
    const photos = ref([]);
    const videos = ref([]);
    
    const takePhoto = async () => {
        await Camera.getPhoto();
    };
    
    const recordVideo = async () => {
        await Camera.recordVideo()
            .maxDuration(60);
    };
    
    const pickFromGallery = async () => {
        await Camera.pickImages()
            .all()
            .multiple();
    };
    
    const handlePhotoTaken = (payload) => {
        photos.value.push(payload.path);
    };
    
    const handleVideoRecorded = (payload) => {
        videos.value.push(payload.path);
    };
    
    const handleMediaSelected = (payload) => {
        if (payload.success) {
            payload.files.forEach(file => {
                if (file.type === 'image') {
                    photos.value.push(file.path);
                } else {
                    videos.value.push(file.path);
                }
            });
        }
    };
    
    onMounted(() => {
        On(Events.Camera.PhotoTaken, handlePhotoTaken);
        On(Events.Camera.VideoRecorded, handleVideoRecorded);
        On(Events.Gallery.MediaSelected, handleMediaSelected);
    });
    
    onUnmounted(() => {
        Off(Events.Camera.PhotoTaken, handlePhotoTaken);
        Off(Events.Camera.VideoRecorded, handleVideoRecorded);
        Off(Events.Gallery.MediaSelected, handleMediaSelected);
    });
    </script>
    
    <template>
        <div>
            <button @click="takePhoto">Take Photo</button>
            <button @click="recordVideo">Record Video</button>
            <button @click="pickFromGallery">Pick from Gallery</button>
            
            <div v-for="photo in photos" :key="photo">
                <img :src="photo" alt="Captured photo" />
            </div>
            
            <div v-for="video in videos" :key="video">
                <video :src="video" controls></video>
            </div>
        </div>
    </template>
    ```
  </Tab>
</Tabs>
